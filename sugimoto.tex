% command to print the name, location, and duration (#2, #3, #4 respectively) of an activity

\newcommand{\datedsubsection}[4][]{
    \subsection*{
        \textbf{\large #2}, #3 \hfill #4
    } % unnumbered subsection
    \vspace{-4pt} % remove vertical space
    #1 % optional subtitle
}

% command to print a subsection title and body (#1 and #2 respectively)
\newcommand{\simplesubsection}[2]{
    \subsection*{\textbf{#1}} % unnumbered subsection
    \vspace{-4pt} % remove vertical space
    #2 % body
}

\newcommand{\name}[1]{

}

% with left-aligned content but without left and right padding

    \begin{tabularx}
          {\textwidth}
          {@{}>{\raggedright\arraybackslash}p{4.4cm}
              !{\color{\primarycolor}\vrule} % primary-colored separator
              >{\raggedright\arraybackslash}X@{}}
        \vspace{-10pt} #1 & \vspace{-10pt} #2
    \end{tabularx}}
}

% command to print skill name and bar given name and number (#1 and #2)
\newcommand{\skill}[2]{
    #1 \hfill \progressbar{#2}
}
